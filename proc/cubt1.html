    subroutine cubt1(nx,p,mx,q,ix,dxm,dx,dxp,dxpp)

    implicit none

    integer :: mx,ix(mx),nx,i,ii
    real(wp) :: p(nx),q(mx),dxm(mx),dx(mx),dxp(mx),dxpp(mx)

    do ii=1,mx
        i = ix(ii)
        q(ii) = dxm(ii)*p(i-1)+dx(ii)*p(i)+dxp(ii)*p(i+1)+dxpp(ii)*p(i+2)
    end do

    end subroutine cubt1
