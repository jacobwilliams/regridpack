    subroutine lint1(nx,p,mx,q,ix,dx)

    implicit none

    integer :: mx,ix(mx),nx,ii,i
    real(wp) :: p(nx),q(mx),dx(mx)

    do ii=1,mx
        i = ix(ii)
        q(ii) = p(i)+dx(ii)*(p(i+1)-p(i))
    end do

    end subroutine lint1
